### **3. Newsletter Signup** (JS)

```javascript
// Add to localStorage
function newsletterSubscribe(email) {
  const subs = JSON.parse(localStorage.getItem('asf-newsletter') || '[]');
  subs.push({email, date: new Date().toISOString()});
  localStorage.setItem('asf-newsletter', JSON.stringify(subs));

  // Send to MailerLite API (free)
  fetch('https://api.mailerlite.com/api/v2/subscribers', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-MailerLite-ApiKey': 'YOUR_API_KEY'
    },
    body: JSON.stringify({
      email: email,
      name: 'Subscriber ASF'
    })
  });

  alert('✅ Inscrito! Você receberá dicas toda semana.');
}
```
